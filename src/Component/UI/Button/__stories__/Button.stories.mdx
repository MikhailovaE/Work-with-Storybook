import { Canvas, Meta, Story } from "@storybook/addon-docs";
import Button from "../Button";
import ButtonGroup from '../ButtonGroup'

<Meta title="Component/Button" component={Button} />

## Variants
You can use 2 variants of buttons:

- primary (default)

- secondary

<Canvas>
  <Story name="Variants">
    {() => {
      return (
        <>
            <ButtonGroup>
                <Button> Primary </Button>
                <Button variant="secondary"> Secondary </Button>
            </ButtonGroup>
        </>
      );
    }}
  </Story>
</Canvas>

## Disabled

If you need blocked button, use ```isDisabled``` props.

<Canvas>
  <Story name="Disabled">
    {() => {
      return (
        <>
            <ButtonGroup>
                <Button isDisabled> Primary </Button>
                <Button variant="secondary" isDisabled> Secondary </Button>
            </ButtonGroup>
        </>
      );
    }}
  </Story>
</Canvas>

## Sizes

Use prop ```size=...``` to change button's size.

- sm

- md (default)

- xl

<Canvas>
  <Story name="Sizes">
    {() => {
      return (
        <>
            <ButtonGroup>
                <Button size='sm'> sm </Button>
            </ButtonGroup>
            <ButtonGroup>
                <Button> md </Button>
            </ButtonGroup>
            <ButtonGroup>
                <Button size='xl'> xl</Button>
            </ButtonGroup>
        </>
      );
    }}
  </Story>
</Canvas>
